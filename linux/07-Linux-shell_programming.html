#!/bin/bash
# 磁盘监控脚本
THRESHOLD=80
disk=$(df -h / | awk 'NR==2{print $5}' | cut -d'%' -f1)

if [ "$disk" -gt "$THRESHOLD" ]; then
  echo "🚨 超过${THRESHOLD}%！"
else
  echo "✅ 正常"
fi

适用：批量处理

🎯 for 循环三种写法

写法一：列表

for ip in 1.1 1.2
do
  ping $ip
done

写法二：seq（推荐）

for i in $(seq 1 100)
do
  echo $i
done

写法三：C风格

for ((i=1;i<=10;i++))
do
  echo $i
done

🌐 进阶：网络连通性检测

15 / 22
综合应用 进阶

📋 任务

检测 192.168.1.1~100 的连通状态。

🔧 要点

  • ping -c 1 -W 1
  • $? 返回值判断
  • & 后台执行

📝 脚本

#!/bin/bash
echo "检测 192.168.1.x..."

for i in $(seq 1 100)
do
  ip="192.168.1.$i"
  ping -c1 -W1 $ip >/dev/null 2>&1
  if [ $? -eq 0 ]; then
    echo "✅ $ip 在线"
  else
    echo "❌ $ip 离线"
  fi
done

🎯 用到了所有知识：变量+判断+循环+命令替换+管道+重定向
